    function showNotification(message, type = "success") {
      const toast = document.getElementById('notification-toast');
      const msgSpan = document.getElementById('notification-message');
      const icon = document.getElementById('notification-icon');

      msgSpan.textContent = message;
      
      if (type === "error") {
        toast.className = "fixed top-4 right-4 z-50 flex items-center gap-2 px-4 py-3 rounded-lg shadow-lg text-white bg-rose-600 transition-transform duration-300";
        icon.className = "fa-solid fa-circle-exclamation";
      } else if (type === "warning") {
        toast.className = "fixed top-4 right-4 z-50 flex items-center gap-2 px-4 py-3 rounded-lg shadow-lg text-white bg-amber-500 transition-transform duration-300";
        icon.className = "fa-solid fa-triangle-exclamation";
      } else {
        toast.className = "fixed top-4 right-4 z-50 flex items-center gap-2 px-4 py-3 rounded-lg shadow-lg text-white bg-emerald-600 transition-transform duration-300";
        icon.className = "fa-solid fa-circle-check";
      }
      
      toast.style.transform = "translateY(0)";
      setTimeout(() => {
        toast.style.transform = "translateY(-150%)";
      }, 4000);
    }

    function generateLetterNumber(tipe) {
      const totalExisting = state.suratKeluar.filter(s => s.tipe === tipe).length + 1;
      const padNumber = String(totalExisting).padStart(3, '0');
      let code = "UND";
      if (tipe === "surat mutasi siswa") code = "MUT";
      else if (tipe === "surat keterangan aktif siswa") code = "KET";
      else if (tipe === "surat dinas") code = "DIN";
      else if (tipe === "surat kelakuan baik") code = "SKKB";
      else if (tipe === "surat kelulusan") code = "LUL";
      
      return `${padNumber}/SDIT-AN/${code}/VI/2026`;
    }

    function deleteSiswa(id) {
      requestConfirm(
        "Hapus Data Siswa",
        "Apakah Anda yakin ingin menghapus data siswa ini? Semua riwayat nilai rapor terkait juga akan ikut terhapus.",
        () => {
          state.siswa = state.siswa.filter(s => s.id !== id);
          state.nilai = state.nilai.filter(n => n.siswaId !== id);
          showNotification("Data siswa berhasil dihapus.", "success");
          renderContent();
        }
      );
    }

    function downloadSiswaFormatCSV() {
      const headers = ["Nama Lengkap", "Gender", "NIS", "NISN", "Tempat Lahir", "Tanggal Lahir", "Nama Ayah", "Nama Ibu", "Kelas", "Status", "Tahun Ajaran"];
      const sampleRow = ["Dewi Lestari", "Perempuan", "222301015", "0129998811", "Karawang", "2016-03-09", "Supardi", "Sari", "Kelas 4A", "Aktif", "2024/2025"];
      
      const csvContent = "\uFEFF" + [headers.join(";"), sampleRow.join(";")].join("\n");
      const blob = new Blob([csvContent], { type: 'text/csv;charset=utf-8;' });
      const url = URL.createObjectURL(blob);
      const link = document.createElement("a");
      link.setAttribute("href", url);
      link.setAttribute("download", "Format_Impor_Siswa_SDIT_Annihayah.csv");
      document.body.appendChild(link);
      link.click();
      document.body.removeChild(link);
      showNotification("Format excel data induk siswa berhasil diunduh.");
    }

    function updateNilaiSelector(type, val) {
      if (type === 'ta') state.selectedTahunNilai = val;
      if (type === 'kelas') {
        state.selectedKelasNilai = val;
        state.selectedSiswaNilai = ""; 
      }
      if (type === 'sem') state.selectedSemester = val;
      if (type === 'siswa') state.selectedSiswaNilai = val;
      
      renderNilaiTab(document.getElementById('main-content'));
    }

    function saveNilaiForm(e) {
      e.preventDefault();
      const form = e.target;
      const mapels = getActiveSubjects(state.selectedTahunNilai);
      
      const mapelValues = {};
      mapels.forEach(m => {
        mapelValues[m.id] = parseFloat(form[`mapel-${m.id}`].value) || 0;
      });

      const dataBaru = {
        id: `N-${state.selectedSiswaNilai}-${state.selectedSemester}`,
        siswaId: state.selectedSiswaNilai,
        kelas: state.selectedKelasNilai,
        semester: state.selectedSemester,
        tahunAjaran: state.selectedTahunNilai,
        mapel: mapelValues,
        absensi: {
          sakit: parseInt(form['abs-sakit'].value) || 0,
          izin: parseInt(form['abs-izin'].value) || 0,
          alfa: parseInt(form['abs-alfa'].value) || 0
        },
        catatan: form['catatan'].value
      };

      const idx = state.nilai.findIndex(n => 
        n.siswaId === state.selectedSiswaNilai &&
        n.kelas === state.selectedKelasNilai &&
        n.semester === state.selectedSemester &&
        n.tahunAjaran === state.selectedTahunNilai
      );

      if (idx !== -1) {
        state.nilai[idx] = dataBaru;
      } else {
        state.nilai.push(dataBaru);
      }

      showNotification("Nilai rapor berhasil disimpan.", "success");
      renderNilaiTab(document.getElementById('main-content'));
    }

    function imporNilaiMasal() {
      const form = document.querySelector('form');
      const mapels = getActiveSubjects(state.selectedTahunNilai);
      mapels.forEach(m => {
        const input = form[`mapel-${m.id}`];
        if (input) input.value = Math.floor(Math.random() * (95 - 75 + 1) + 75);

      form['abs-sakit'].value = 1;
      form['abs-izin'].value = 0;
      form['abs-alfa'].value = 0;
      form['catatan'].value = "Sangat rajin dan memiliki potensi akademik yang sangat baik.";
      showNotification("Berhasil mensimulasikan impor data nilai Excel masal.", "success");
    }

    function downloadLegerFormat() {
      const activeSubjs = getActiveSubjects(state.selectedTahunNilai);
      const headers = ["Siswa ID", "Nama Siswa", ...activeSubjs.map(s => s.id), "Sakit", "Izin", "Alfa", "Catatan"];
      
      const filteredSiswaLeger = state.siswa.filter(s => s.kelas === state.selectedKelasNilai && s.status === "Aktif");
      if (filteredSiswaLeger.length === 0) {
        showNotification("Tidak ada siswa aktif di kelas ini untuk dibuat format legernya!", "warning");
        return;
      }

      const rows = filteredSiswaLeger.map(sv => {
        const nRecord = state.nilai.find(n => 
          n.siswaId === sv.id && 
          n.kelas === state.selectedKelasNilai && 
          n.semester === state.selectedSemester &&
          n.tahunAjaran === state.selectedTahunNilai
        );
        const mapelVals = activeSubjs.map(s => nRecord?.mapel?.[s.id] !== undefined ? nRecord.mapel[s.id] : 0);
        return [
          sv.id,
          sv.nama,
          ...mapelVals,
          nRecord?.absensi?.sakit || 0,
          nRecord?.absensi?.izin || 0,
          nRecord?.absensi?.alfa || 0,
          nRecord?.catatan || ""
        ];
      });

      const csvContent = "\uFEFF" + [headers.join(";"), ...rows.map(r => r.join(";"))].join("\n");
      const blob = new Blob([csvContent], { type: 'text/csv;charset=utf-8;' });
      const url = URL.createObjectURL(blob);
      const link = document.createElement("a");
      link.setAttribute("href", url);
      link.setAttribute("download", `Leger_${state.selectedKelasNilai.replace(" ", "_")}_Semester_${state.selectedSemester}.csv`);
      document.body.appendChild(link);
      link.click();
      document.body.removeChild(link);
      showNotification("Format Leger Rapor (CSV) berhasil diunduh.");
    }

    function handleImportLegerCSV(e) {
      const file = e.target.files[0];
      if (!file) return;
      const reader = new FileReader();
      reader.onload = (event) => {
        const text = event.target.result;
        const lines = text.split("\n");
        if (lines.length < 2) {
          showNotification("Format file tidak valid!", "error");
          return;
        }
        const headers = lines[0].replace(/\r/g, "").split(";");
        const activeSubjs = getActiveSubjects(state.selectedTahunNilai);
        const mapelIndices = {};
        activeSubjs.forEach(s => {
          mapelIndices[s.id] = headers.indexOf(s.id);
        });
        const sakitIdx = headers.indexOf("Sakit");
        const izinIdx = headers.indexOf("Izin");
        const alfaIdx = headers.indexOf("Alfa");
        const catatanIdx = headers.indexOf("Catatan");

        let updatedNilai = [...state.nilai];
        let importCount = 0;

        for (let i = 1; i < lines.length; i++) {
          if (!lines[i].trim()) continue;
          const cols = lines[i].replace(/\r/g, "").split(";");
          const siswaId = cols[0];
          if (!siswaId) continue;

          const mapelVals = {};
          activeSubjs.forEach(s => {
            const colIdx = mapelIndices[s.id];
            mapelVals[s.id] = colIdx !== -1 && cols[colIdx] !== undefined ? parseFloat(cols[colIdx]) || 0 : 0;
          });

          const sakitVal = sakitIdx !== -1 && cols[sakitIdx] !== undefined ? parseInt(cols[sakitIdx]) || 0 : 0;
          const izinVal = izinIdx !== -1 && cols[izinIdx] !== undefined ? parseInt(cols[izinIdx]) || 0 : 0;
          const alfaVal = alfaIdx !== -1 && cols[alfaIdx] !== undefined ? parseInt(cols[alfaIdx]) || 0 : 0;
          const catatanVal = catatanIdx !== -1 && cols[catatanIdx] !== undefined ? cols[catatanIdx] : "";

          const item = {
            id: `N-${siswaId}-${state.selectedSemester}`,
            siswaId,
            kelas: state.selectedKelasNilai,
            semester: state.selectedSemester,
            tahunAjaran: state.selectedTahunNilai,
            mapel: mapelVals,
            absensi: { sakit: sakitVal, izin: izinVal, alfa: alfaVal },
            catatan: catatanVal
          };

          const idx = updatedNilai.findIndex(n => 
            n.siswaId === siswaId && 
            n.kelas === state.selectedKelasNilai && 
            n.semester === state.selectedSemester && 
            n.tahunAjaran === state.selectedTahunNilai
          );

          if (idx !== -1) {
            updatedNilai[idx] = item;
          } else {
            updatedNilai.push(item);
          }
          importCount++;
        }

        state.nilai = updatedNilai;
        showNotification(`Berhasil mengimpor nilai leger untuk ${importCount} siswa.`);
        renderContent();
      };
      reader.readAsText(file);
      e.target.value = '';
    }

    function syncCptpInput() {
      const kelas = document.getElementById('cptp-kelas').value;
      const mapel = document.getElementById('cptp-mapel').value;
      const record = state.cptp.find(c => c.kelas === kelas && c.mapel === mapel);
      
      document.getElementById('cptp-cp').value = record ? record.cp : "";
      document.getElementById('cptp-tp').value = record ? record.tp : "";
    }

    function saveCptp() {
      const kelas = document.getElementById('cptp-kelas').value;
      const mapel = document.getElementById('cptp-mapel').value;
      const cp = document.getElementById('cptp-cp').value.trim();
      const tp = document.getElementById('cptp-tp').value.trim();

      if (!cp || !tp) {
        showNotification("Kolom CP dan TP wajib diisi!", "error");
        return;
      }

      const idx = state.cptp.findIndex(c => c.kelas === kelas && c.mapel === mapel);
      const data = { kelas, mapel, cp, tp };

      if (idx !== -1) {
        state.cptp[idx] = data;
      } else {
        state.cptp.push(data);
      }

      showNotification(`Berhasil menyimpan target CP & TP.`);
      renderSettingsSubTab();
    }

    function deleteCptp(kelas, mapel) {
      requestConfirm(
        "Hapus CP & TP",
        `Hapus target CP & TP untuk kelas ${kelas} mapel ${mapel}?`,
        () => {
          state.cptp = state.cptp.filter(c => !(c.kelas === kelas && c.mapel === mapel));
          showNotification("Target CP & TP berhasil dihapus.");
          renderSettingsSubTab();
        }
      );
    }

    function saveSettings() {

      state.settings.activeYear = document.getElementById('set-ta-active').value;

      showNotification("Pengaturan berhasil disimpan.", "success");
      switchTab('dashboard');
    }

    function saveSiswaForm(e, id) {
      e.preventDefault();
      const form = e.target;
      const status = form.status.value;

      const baseData = {
        nama: form.nama.value,
        gender: form.gender.value,
        nis: form.nis.value,
        nisn: form.nisn.value,
        tempatLahir: form.tempatLahir.value,
        tanggalLahir: form.tanggalLahir.value,
        namaAyah: form.namaAyah.value,
        namaIbu: form.namaIbu.value,
        kelas: form.kelas.value,
        status: status,
        tahunAjaran: form.tahunAjaran.value
      };

      if (status === 'Lulus') {
        baseData.noIjazah = form.noIjazah.value;
        baseData.tanggalLulus = form.tanggalLulus.value;
        baseData.nilaiRataIjazah = parseFloat(form.nilaiRataIjazah.value) || 0;
      } else if (status === 'Mutasi Masuk') {
        baseData.tanggalMasuk = form.tanggalMasuk.value;
        baseData.sekolahAsal = form.sekolahAsal.value;
        baseData.alamatAsal = form.alamatAsal.value;
      } else if (status === 'Mutasi Keluar') {
        baseData.tanggalKeluar = form.tanggalKeluar.value;
        baseData.sekolahTujuan = form.sekolahTujuan.value;
        baseData.alamatTujuan = form.alamatTujuan.value;
      }

      if (id) {
        const idx = state.siswa.findIndex(x => x.id === id);
        state.siswa[idx] = { ...state.siswa[idx], ...baseData };
        showNotification("Berhasil memperbarui data siswa.");
      } else {
        baseData.id = `S-${Date.now()}`;
        state.siswa.push(baseData);
        showNotification("Berhasil menambah data siswa baru.");
      }

      closeModal();
      renderContent();
    }

    function syncSKTemplate() {
      const studentId = document.getElementById('sk-siswa-select').value;
      const tipe = document.getElementById('sk-tipe-select').value;
      
      document.getElementById('sk-modal-no-preview').textContent = generateLetterNumber(tipe);

      if (!studentId) {
        document.getElementById('sk-penerima').value = "";
        document.getElementById('sk-keterangan').value = "";
        return;
      }

      const s = state.siswa.find(item => item.id === studentId);
      if (!s) return;

      let penerima = "";
      let keterangan = "";

      if (tipe === "surat mutasi siswa") {
        penerima = `Orang Tua / Wali dari ${s.nama}`;
        keterangan = `SURAT MUTASI KELUAR SISWA\n\n` +
          `Nama Lengkap   : ${s.nama}\n` +
          `NIS / NISN     : ${s.nis} / ${s.nisn}\n` +
          `Jenis Kelamin  : ${s.gender}\n` +
          `Kelas Terakhir : ${s.kelas}\n\n` +
          `Telah resmi mengajukan permohonan kepindahan (mutasi keluar) per tanggal ${s.tanggalKeluar || state.settings.printDateLetter} ` +
          `menuju sekolah tujuan ${s.sekolahTujuan || '[Sekolah Tujuan]'} yang beralamat di ${s.alamatTujuan || '[Alamat Tujuan]'} ` +
          `dikarenakan mengikuti domisili orang tua.`;
      } else if (tipe === "surat keterangan aktif siswa") {
        penerima = `Yth. Pimpinan Instansi / Wali Murid ${s.nama}`;
        keterangan = `SURAT KETERANGAN AKTIF BELAJAR\n\n` +
          `Nama Lengkap   : ${s.nama}\n` +
          `NIS / NISN     : ${s.nis} / ${s.nisn}\n` +
          `Kelas Saat Ini : ${s.kelas}\n` +
          `Tahun Ajaran   : ${s.tahunAjaran || state.settings.activeYear}\n\n` +
          `Menerangkan dengan sesungguhnya bahwa nama siswa tersebut di atas adalah benar-benar siswa aktif belajar yang terdaftar ` +
          `di SDIT Annihayah pada tahun pelajaran berjalan, dan memiliki catatan kehadiran serta akademis yang baik.`;
      } else if (tipe === "surat kelulusan") {
        penerima = `Orang Tua / Wali dari ${s.nama}`;
        keterangan = `SURAT KETERANGAN KELULUSAN\n\n` +
          `Nama Lengkap   : ${s.nama}\n` +
          `NIS / NISN     : ${s.nis} / ${s.nisn}\n` +
          `No. Ijazah     : ${s.noIjazah || '[Belum Terbit / Dalam Proses]'}\n` +
          `Rata-Rata Nilai: ${s.nilaiRataIjazah || '-'}\n\n` +
          `Berdasarkan kriteria kelulusan satuan pendidikan SDIT Annihayah Tahun Pelajaran ${s.tahunAjaran || state.settings.activeYear}, ` +
          `yang bersangkutan dinyatakan LULUS dari jenjang pendidikan Sekolah Dasar per tanggal ${s.tanggalLulus || state.settings.printDateTranscript}.`;
      } else if (tipe === "surat kelakuan baik") {

        keterangan = `SURAT KETERANGAN KELAKUAN BAIK (SKKB)\n\n` +
          `Nama Lengkap   : ${s.nama}\n` +
